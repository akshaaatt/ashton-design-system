# Ashton Design System

## Install

```bash
npm install --save ashton-design-system
```

or

```bash
yarn add ashton-design-system
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'ashton-design-system'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
